43  ReturnCodes MXNetServable::SetBatchSize(const int &new_size) {
44  std::lock_guard<std::mutex> guard_input (input_mutex_);
45 
46  if (new_size <= current_n_) {
47  return ReturnCodes::NEXT_BATCH;
48  }
49 
50  // Reshape the input
51  input_shape_ = mx::Shape(new_size, input_shape_[1], input_shape_[2], input_shape_[3]);

62  ReturnCodes MXNetServable::AddToBatch(const TensorMessage &message) {
63 
64  const std::string &client_id = message.client_id();
65  std::vector<float> message_data (message.buffer().data(), message.buffer().data() + message.buffer_size());
66 
67  if (!bind_called_) {
68  return ReturnCodes::NEED_BIND_CALL;
69  }
70 
71  if (message.n() > input_shape_[0]) {
72  return ReturnCodes::BATCH_TOO_LARGE;
73  }
74 
75  if (message.k() != input_shape_[1] || message.nr() != input_shape_[2] || message.nc() != input_shape_[3]) {
76  return ReturnCodes::SHAPE_INCORRECT;
77  }
78 
79  {
80 
81  std::lock_guard<std::mutex> guard_input(input_mutex_);
82 
83  if (message.n() + current_n_ > input_shape_[0]) {
84  ProcessCurrentBatch_();
85  return ReturnCodes::NEXT_BATCH;
86  }
87 
88  result_by_client_.erase(client_id); // clears room for the new result
89 
90  if (idx_by_client_.find(client_id) == idx_by_client_.end()) {
91  idx_by_client_[client_id] = std::make_pair(current_n_, current_n_ + message.n());
92  } else {
93  idx_by_client_[client_id].second += message.n();
94  }
95 
96  current_batch_.emplace_back(
97  mx::NDArray(message.buffer().data(),
98  mx::Shape(message.n(), input_shape_[1], input_shape_[2], input_shape_[3]), ctx_)
99  );
100 
101  current_n_ += message.n();
102  if (current_n_ <= input_shape_[0]) {
103  if (current_n_ == input_shape_[0]) {
104  ProcessCurrentBatch_();
105  }
106  }
107 
108  }
109 
110  return ReturnCodes::OK;
111  }

113  ReturnCodes MXNetServable::GetResult(const std::string &client_id, TensorMessage *message) {
114 
115  std::unique_lock<std::mutex> lk(result_mutex_);
116  result_cv_.wait(lk, [&, this](){ // This will block forever if the client only calls GetResult and never AddToBatch
117  auto done = done_processing_by_client_.find(client_id);
118  if (done != done_processing_by_client_.end()) {
119  done = done_processing_by_client_.erase(done);
120  return true;
121  }
122  else {
123  return false;
124  }
125  });
126 
127  auto result = result_by_client_.find(client_id);
128  mx::NDArray &result_array = result->second;
129 
130  std::vector<mx_uint> result_shape = result_array.GetShape();
131 
132  google::protobuf::RepeatedField<float> data(result_array.GetData(), result_array.GetData()+result_array.Size());
133  message->mutable_buffer()->Swap(&data);
134  result = result_by_client_.erase(result);
135 
136 
137  message->set_n(result_shape[0]);
138  message->set_k(result_shape[1]);
139  message->set_nr(result_shape[2]);
140  message->set_nc(result_shape[3]);
141  message->set_client_id(client_id);
142 
143  return ReturnCodes::OK;
144

147  ReturnCodes MXNetServable::Bind(BindArgs &args) {
148  bind_called_ = true;
149 
150  try {
151  RawBindArgs &raw_args = dynamic_cast<RawBindArgs &>(args);
152  servable_ = raw_args.net;
153  LoadParameters_(raw_args.parameters);
154  BindExecutor_();
155  return ReturnCodes::OK;
156  }
157  catch (std::bad_cast &e) {
158  ;
159  }
160 
161  try {
162  FileBindArgs &file_args = dynamic_cast<FileBindArgs &>(args);
163  servable_ = mx::Symbol::Load(file_args.symbol_filename);
164  std::map<std::string, mx::NDArray> parameters = mx::NDArray::LoadToMap(file_args.parameters_filename);
165  LoadParameters_(parameters);
166  BindExecutor_();
167  return ReturnCodes::OK;
168  }
169  catch (std::bad_cast &e) {
170  ;
171  }
172 
173  return ReturnCodes::NO_SUITABLE_BIND_ARGS;
174

188  void MXNetServable::LoadParameters_(std::map<std::string, mx::NDArray> &parameters) {
189  for (const auto &k : parameters) {
190  if (k.first.substr(0, 4) == "aux:") {
191  auto name = k.first.substr(4, k.first.size() - 4);
192  aux_map_[name] = k.second.Copy(ctx_);
193  }
194  if (k.first.substr(0, 4) == "arg:") {
195  auto name = k.first.substr(4, k.first.size() - 4);
196  args_map_[name] = k.second.Copy(ctx_);
197  }

203  void MXNetServable::ProcessCurrentBatch_() {
204 
205  mx::Operator("concat")(current_batch_)
206  .SetParam("dim", 0)
207  .SetParam("num_args", current_batch_.size())
208  .Invoke(args_map_["data"]);
209 
210  executor_->Forward(false);
211 
212  mx::NDArray &result = executor_->outputs[0];
213  mx::NDArray::WaitAll();
214 
215  for (auto &client_idx: idx_by_client_) {
216  int client_batch_size = client_idx.second.second - client_idx.second.first;
217  result_by_client_[client_idx.first] = mx::NDArray(mx::Shape(client_batch_size, output_shape_[1]), ctx_);
218  result_by_client_[client_idx.first] = 0.f;
219  result_by_client_[client_idx.first] += result.Slice(client_idx.second.first, client_idx.second.second);
220  done_processing_by_client_.emplace(client_idx.first);
221  }
222 
223  idx_by_client_.clear();
224 
225  // Reset everyone
226  current_batch_.clear();
227  result_cv_.notify_all();
228  current_n_ = 0;
229 
230  }
