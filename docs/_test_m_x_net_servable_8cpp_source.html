36  mx::Symbol SimpleSymbolFactory(int n_hidden) {
37  mx::Symbol data = mx::Symbol::Variable("data");
38  mx::Symbol m = mx::Symbol::Variable("m");
39  mx::Symbol b = mx::Symbol::Variable("b");

46  Serving::TensorMessage ToMessage(mx::NDArray &arr) {
47  std::vector<mx_uint> result_shape = arr.GetShape();
48 
49  Serving::TensorMessage message;
50  google::protobuf::RepeatedField<float> data(arr.GetData(), arr.GetData()+arr.Size());
51  message.mutable_buffer()->Swap(&data);
52 
53  message.set_n(result_shape[0]);
54  message.set_k(result_shape[1]);
55  message.set_nr(result_shape[2]);
56  message.set_nc(result_shape[3]);
57  message.set_client_id("data");
58 
59  return message;
60  }
