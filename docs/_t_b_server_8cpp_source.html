73  grpc::Status TBServer::Process(ServerContext *ctx, const TensorMessage *req, TensorMessage *rep) {
74 
75  auto user = users_.find(req->client_id());
76  if (user == users_.end()) {
77  grpc::Status early_exit_status (grpc::FAILED_PRECONDITION, "Connect not called, client id unknown");
78  return early_exit_status;
79  }
80 
81  // TODO: make sure that this is all going to the same servable - only one instance!
82  ReturnCodes code = servable_->AddToBatch(*req); // Add to batch and move to the next stage
83 
84  switch (code) {
85  case OK: break;
86  case NEED_BIND_CALL: {
87  grpc::Status early_exit_status (grpc::FAILED_PRECONDITION, "Bind not called on servable");
88  return early_exit_status;
89  }
90  case SHAPE_INCORRECT: {
91  grpc::Status early_exit_status (grpc::INVALID_ARGUMENT, "Input tensor shape incorrect");
92  return early_exit_status;
93  }
94  case NEXT_BATCH: {
95  grpc::Status early_exit_status (grpc::UNAVAILABLE, "Attempted to add to already full batch");
96  return early_exit_status;
97  }
98  case BATCH_TOO_LARGE: {
99  grpc::Status early_exit_status (grpc::INVALID_ARGUMENT, "Batch request was too large, split into smaller pieces and retry");
100  return early_exit_status;
101  }
102  case NO_SUITABLE_BIND_ARGS: break; // this one won't be thrown by the function
103  }
104 
105  code = servable_->GetResult(req->client_id(), rep);
106 
107  switch (code) {
108  case OK: break;
109  case NEXT_BATCH: {
110  grpc::Status early_exit_status (grpc::UNAVAILABLE, "Try again later, processing hasn't yet started!");
111  return early_exit_status;
112  }
113  }
114 
115  return grpc::Status::OK;
116  }
